React.js React jest darmową i open-source'ową biblioteką front-end JavaScript służącą do budowania interfejsów użytkownika opartych na komponentach UI. Jest utrzymywana przez Meta oraz społeczność indywidualnych deweloperów i firm. 
        <div id="like_button_container"></div>
<script type="text/babel">
  const Button = () => {
    const [liked, setLiked] = React.useState(false);
    if (liked) return "You liked this.";
    return <button onClick={() => setLiked(true)}>Like</button>;
  };
  const domContainer = document.querySelector(
    "#like_button_container"
  );
  ReactDOM.render(<Button />, domContainer);
</script>
